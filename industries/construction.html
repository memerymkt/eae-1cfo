GitHub Issues and Projects for equipment maintenance scheduling with financial cost tracking. Preventive maintenance costs and breakdown repair expenses monitored.

# GitHub Actions workflow for construction equipment tracking
name: Equipment Financial Tracking
on:
  schedule:
    - cron: '0 8 * * 1'  # Weekly on Monday
  workflow_dispatch:

jobs:
  equipment-analysis:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Analyze Equipment Utilization
        run: |
          # Analyze equipment usage data
          echo "Analyzing equipment utilization..."
          
          # Calculate utilization rates
          # Check maintenance schedules
          # Monitor equipment costs
          
          # Generate utilization report
          cat > equipment-report.md << EOF
          # Weekly Equipment Utilization Report
          
          ## Fleet Overview
          - Total Equipment: 25 units
          - Active Equipment: 20 units
          - Average Utilization: 78%
          
          ## Maintenance Schedule
          - Scheduled Maintenance: 3 units this week
          - Preventive Maintenance Cost: $12,500
          - Breakdown Repair Cost: $8,200
          
          ## Financial Impact
          - Equipment Cost per Hour: $85
          - Utilization Revenue: $156,000
          - Maintenance Cost Ratio: 8.2%
          
          ## Recommendations
          - Schedule maintenance for Unit #7
          - Review utilization for bulldozer fleet
          - Consider equipment upgrade for excavator
          EOF
          
          echo "equipment-report.md" >> .gitignore

  financial-integration:
    runs-on: ubuntu-latest
    steps:
      - name: Update Financial Dashboard
        run: |
          # Send equipment data to financial system
          curl -X POST ${{ secrets.FINANCIAL_API_URL }}/equipment \
            -H "Authorization: Bearer ${{ secrets.FINANCIAL_API_KEY }}" \
            -d "{
              \"report_date\": \"$(date +%Y-%m-%d)\",
              \"utilization_rate\": 78,
              \"maintenance_cost\": 8200,
              \"revenue_impact\": 156000
            }"
